// count the digit d in the squares of numbers up to n
function nbDig(n, d) {
  let stringOfSquares = '';
  let count = 0;
  for (let i = 0; i <= n; i++) {
    stringOfSquares += i**2;
  }
  for (let k = 0; k < stringOfSquares.length; k++) {
    if (stringOfSquares[k] == d) {
      count++;
    }
  }
  return count;
}
